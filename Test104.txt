		//향상된 Bubble Sort 구현

		int pass=0;
		boolean flag;
		do
		{
			flag=false;
			pass++;
			for (int i=0 ; i<a.length-pass ; i++ )
			{
				if (a[i]> a[i+1])
				{
					//자리 바꾸기
